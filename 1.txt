function offreExpiree(dates) {
  if(!dates||dates.length<MAX) return false;
  const [j,m,a]=dates[MAX-1].split('/');
  const diff=(new Date()-new Date(a,m-1,j))/(1000*60*60*24);
  return diff>DELAI;
}

function joursRestants(dates) {
  if(!dates||dates.length<MAX) return 0;
  const [j,m,a]=dates[MAX-1].split('/');
  return Math.max(0,Math.ceil(DELAI-(new Date()-new Date(a,m-1,j))/(1000*60*60*24)));
}

function renderTampons(cid, visites, dates=[]) {
  const c=document.getElementById(cid); c.innerHTML='';
  for(let i=0;i<=MAX;i++) {
    const w=document.createElement('div'); w.className='tampon-wrap';
    const d=document.createElement('div');
    d.className='tampon'+(i<visites?' filled':'');
    d.textContent=i===MAX?'🎁':'✂️';
    w.appendChild(d);
    const dt=document.createElement('div');
    dt.className='tampon-date';
    dt.textContent=i<visites&&dates[i]?dates[i]:'';
    w.appendChild(dt);
    c.appendChild(w);
  }
}

function afficherCarte(d) {
  document.getElementById('c-nom').textContent=d.prenom;
  document.getElementById('c-tel').textContent='📞 '+d.tel;
  document.getElementById('c-date-creation').textContent='📅 Client depuis le '+d.created;
  const v=d.visites||0, dates=d.dates||[];
  document.getElementById('c-progress').textContent=v+' / '+MAX;
  renderTampons('c-tampons',v,dates);
  const expire=offreExpiree(dates), offert=v>=MAX&&!expire;
  document.getElementById('c-offert').style.display=offert?'block':'none';

  if(offert) document.getElementById('c-offert-date').textContent='⏰ Il te reste '+joursRestants(dates)+' jour(s) !';
}

  document.getElementById('load-rech').style.display='none';
};

window.trouverClient = async () => {
  const tel=normTel(document.getElementById('i-gtel').value.trim());
  const err=document.getElementById('err-ger');
  if(!tel){err.textContent='Entre un numéro.';err.style.display='block';return;}

  document.getElementById('load-ger').style.display='block';

    else{clientCur={...snap.data(),id:tel};afficherGerant(clientCur);}
  } catch(e){err.textContent='Erreur de connexion.';err.style.display='block';}
  document.getElementById('load-ger').style.display='none';
};

window.chargerClients = async () => {
  go('s-liste');
  document.getElementById('load-liste').style.display='block';
  document.getElementById('liste-clients').innerHTML='';
  try {
    const snap=await getDocs(collection(db,'clients'));
    const clients=[];
    snap.forEach(d=>clients.push({...d.data(),id:d.id}));
    clients.sort((a,b)=>((b.visites||0)-(a.visites||0)));

    // Stats
    const total=clients.length;
    const offerts=clients.filter(c=>c.visites>=MAX&&!offreExpiree(c.dates||[])).length;
    const expires=clients.filter(c=>c.visites>=MAX&&offreExpiree(c.dates||[])).length;
    document.getElementById('stats-row').innerHTML=`
      <div class="stat-box"><div class="stat-num">${total}</div><div class="stat-label">Clients</div></div>
      <div class="stat-box"><div class="stat-num" style="color:#4ade80">${offerts}</div><div class="stat-label">Offres dispo</div></div>
      <div class="stat-box"><div class="stat-num" style="color:#ff6b6b">${expires}</div><div class="stat-label">Expirées</div></div>
    `;

    const liste=document.getElementById('liste-clients');
    if(clients.length===0){liste.innerHTML='<p style="text-align:center;color:var(--muted);padding:20px">Aucun client encore</p>';return;}

    clients.forEach(c=>{
      const v=c.visites||0, dates=c.dates||[];
      const expire=offreExpiree(dates), offert=v>=MAX&&!expire;
      let badgeClass='badge', badgeText=v+'/'+MAX;
      if(offert){badgeClass='badge badge-offert';badgeText='🎁 Offerte !';}
      else if(v>=MAX&&expire){badgeClass='badge badge-expire';badgeText='⏰ Expirée';}
      const derniere=dates.length?'Dernière visite : '+dates[dates.length-1]:'Pas encore venu';
      const row=document.createElement('div');
      row.className='client-row';
      row.innerHTML=`
        <div class="client-row-left">
          <div class="nom">${c.prenom}</div>
          <div class="sub">📞 ${c.tel}</div>
          <div class="sub">📅 ${derniere}</div>
        </div>
        <div class="client-row-right"><span class="${badgeClass}">${badgeText}</span></div>
      `;
      row.onclick=()=>{
        clientCur=c;
        document.getElementById('i-gtel').value=c.tel;
        afficherGerant(c);
        go('s-gerant-client');
      };
      liste.appendChild(row);
    });
  } catch(e){console.error(e);}
  document.getElementById('load-liste').style.display='none';
};

window.demanderPin = action => {
  pinAction=action;
  const t={gerant:'Espace Gérant',tampon:'Valider la visite',reset:'Coupe offerte',supprimer:'Supprimer client'};
  document.getElementById('pin-title').textContent=t[action]||'Code';
  document.getElementById('pin-inp').value='';
  document.getElementById('pin-err').style.display='none';
  document.getElementById('pin-ov').classList.add('open');
};

window.ferPin = ()=>{document.getElementById('pin-ov').classList.remove('open');pinAction=null;};

window.valPin = async () => {
  if(document.getElementById('pin-inp').value!==PIN){
    document.getElementById('pin-err').style.display='block';return;
  }
  const action=pinAction; ferPin();
  if(action==='gerant'){go('s-gerant');return;}
  if(!clientCur) return;
  try {
    if(action==='tampon'){
      const newV=(clientCur.visites||0)+1;
      const newDates=[...(clientCur.dates||[]),today()];
      await updateDoc(doc(db,'clients',clientCur.id),{visites:newV,dates:newDates});
      clientCur.visites=newV; clientCur.dates=newDates;
      const ok=document.getElementById('msg-ok-ger');
      ok.style.display='block'; setTimeout(()=>ok.style.display='none',2000);
      afficherGerant(clientCur);
    } else if(action==='reset'){
      await updateDoc(doc(db,'clients',clientCur.id),{visites:0,dates:[]});
      clientCur.visites=0; clientCur.dates=[];
      const ok=document.getElementById('msg-reset-ger');
      ok.style.display='block'; setTimeout(()=>ok.style.display='none',2000);
      afficherGerant(clientCur);
    } else if(action==='supprimer'){
      await deleteDoc(doc(db,'clients',clientCur.id));
      clientCur=null;
      document.getElementById('g-info').style.display='none';
      document.getElementById('i-gtel').value='';
      go('s-gerant');
    }
  } catch(e){console.error(e);}
};

document.getElementById('pin-inp').addEventListener('keyup',e=>{if(e.key==='Enter')window.valPin();});
