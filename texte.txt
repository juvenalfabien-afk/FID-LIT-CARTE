const app = initializeApp(firebaseConfig);

const PIN = "1141";
const MAX = 6; // 4 visites + 5ème offerte
const DELAI_JOURS = 30;
let pinAction = null, clientCur = null;

const normTel = t => t.replace(/[\s\-\.]/g,'');
const today = () => new Date().toLocaleDateString('fr-FR');

// Vérifie si offre expirée (> 15 jours depuis le 4ème tampon)
function offreExpiree(dates) {
  if (!dates || dates.length < MAX) return false;
  const derniere = dates[MAX-1];
  if (!derniere) return false;
  const [j,m,a] = derniere.split('/');
  const dateTampon = new Date(a, m-1, j);
  const diff = (new Date() - dateTampon) / (1000*60*60*24);
  return diff > DELAI_JOURS;
}

function joursRestants(dates) {
  if (!dates || dates.length < MAX) return 0;
  const [j,m,a] = dates[MAX-1].split('/');
  const dateTampon = new Date(a, m-1, j);
  const diff = Math.ceil(DELAI_JOURS - (new Date() - dateTampon) / (1000*60*60*24));
  return Math.max(0, diff);
}

window.go = id => {
  document.querySelectorAll('.screen').forEach(s=>s.classList.remove('active'));
  document.getElementById(id).classList.add('active');
};

function renderTampons(containerId, visites, dates=[]) {
  const c = document.getElementById(containerId); c.innerHTML='';
  for(let i=0;i<=MAX;i++) {
    const w=document.createElement('div'); w.className='tampon-wrap';
    const d=document.createElement('div');
    d.className='tampon'+(i<visites?' filled':'');
    d.textContent=i===MAX?'🎁':'✂️';
    w.appendChild(d);
    const dt=document.createElement('div');
    dt.className='tampon-date';
    dt.textContent=i<visites&&dates[i]?dates[i]:'';
    w.appendChild(dt);
    c.appendChild(w);
  }
}

function afficherCarte(d) {
  document.getElementById('c-nom').textContent=d.prenom;
  document.getElementById('c-tel').textContent=d.tel;
  const v=d.visites||0;
  const dates=d.dates||[];
  document.getElementById('c-progress').textContent=v+' / '+MAX;
  renderTampons('c-tampons',v,dates);
  const expire=offreExpiree(dates);
  const offert=v>=MAX&&!expire;
  document.getElementById('c-offert').style.display=offert?'block':'none';
  document.getElementById('c-alerte').style.display=(v>=MAX&&expire)?'block':'none';
  if(offert){
    const jr=joursRestants(dates);
    document.getElementById('c-offert-date').textContent='⏰ Il te reste '+jr+' jour(s) pour profiter de ta coupe offerte !';
  }
}

function afficherGerant(d) {
  document.getElementById('g-nom').textContent=d.prenom;
  document.getElementById('g-tel-d').textContent=d.tel;
  const v=d.visites||0;
  const dates=d.dates||[];
  document.getElementById('g-progress').textContent=v+' / '+MAX;
  renderTampons('g-tampons',v,dates);
  const expire=offreExpiree(dates);
  const offert=v>=MAX&&!expire;
  document.getElementById('g-info').style.display='block';
  document.getElementById('g-alerte').style.display=(v>=MAX&&expire)?'block':'none';
  document.getElementById('btn-stamp').style.display=(v>=MAX)?'none':'flex';
  document.getElementById('btn-reset').style.display=offert?'flex':'none';
}

window.creerCarte = async () => {
  const prenom=document.getElementById('i-prenom').value.trim();
  const tel=normTel(document.getElementById('i-tel').value.trim());
  const email=document.getElementById('i-email').value.trim();
  const err=document.getElementById('err-ins');
  if(!prenom||!tel){err.textContent='Remplis prénom et téléphone !';err.style.display='block';return;}
  err.style.display='none';
  document.getElementById('load-ins').style.display='block';
  try {
    const ref=doc(db,'clients',tel);
    const snap=await getDoc(ref);
    if(!snap.exists()) await setDoc(ref,{prenom,tel,email,visites:0,dates:[],created:today()});
    const data=(await getDoc(ref)).data();
    clientCur={...data,id:tel};
    afficherCarte(data);
    go('s-carte');
  } catch(e){err.textContent='Erreur Firebase. Vérifie ta connexion.';err.style.display='block';console.error(e);}
  document.getElementById('load-ins').style.display='none';
};

window.rechercherCarte = async () => {
  const tel=normTel(document.getElementById('i-rech').value.trim());
  const err=document.getElementById('err-rech');
  if(!tel){err.textContent='Entre ton numéro.';err.style.display='block';return;}
  err.style.display='none';
  document.getElementById('load-rech').style.display='block';
  try {
    const snap=await getDoc(doc(db,'clients',tel));
    if(!snap.exists()){err.textContent='Aucune carte trouvée.';err.style.display='block';}
    else{clientCur={...snap.data(),id:tel};afficherCarte(snap.data());go('s-carte');}
  } catch(e){err.textContent='Erreur Firebase.';err.style.display='block';}
  document.getElementById('load-rech').style.display='none';
};

window.trouverClient = async () => {
  const tel=normTel(document.getElementById('i-gtel').value.trim());
  const err=document.getElementById('err-ger');
  if(!tel){err.textContent='Entre un numéro.';err.style.display='block';return;}
  err.style.display='none';
  document.getElementById('load-ger').style.display='block';
  try {
    const snap=await getDoc(doc(db,'clients',tel));
    if(!snap.exists()){err.textContent='Client introuvable.';err.style.display='block';document.getElementById('g-info').style.display='none';}
    else{clientCur={...snap.data(),id:tel};afficherGerant(clientCur);}
  } catch(e){err.textContent='Erreur Firebase.';err.style.display='block';}
  document.getElementById('load-ger').style.display='none';
};

window.demanderPin = action => {
  pinAction=action;
  const t={gerant:'Espace Gérant',tampon:'Valider la visite',reset:'Coupe offerte utilisée'};
  document.getElementById('pin-title').textContent=t[action]||'Code';
  document.getElementById('pin-inp').value='';
  document.getElementById('pin-err').style.display='none';
  document.getElementById('pin-ov').classList.add('open');
};

window.ferPin = () => {document.getElementById('pin-ov').classList.remove('open');pinAction=null;};

window.valPin = async () => {
  if(document.getElementById('pin-inp').value!==PIN){
    document.getElementById('pin-err').style.display='block';return;
  }
  const action=pinAction; ferPin();
  if(action==='gerant'){go('s-gerant');return;}
  if(!clientCur) return;
  try {
    if(action==='tampon'){
      const newV=(clientCur.visites||0)+1;
      const newDates=[...(clientCur.dates||[]),today()];
      await updateDoc(doc(db,'clients',clientCur.id),{visites:newV,dates:newDates});
      clientCur.visites=newV; clientCur.dates=newDates;
      const ok=document.getElementById('msg-ok-ger');
      ok.style.display='block'; setTimeout(()=>ok.style.display='none',2000);
    } else if(action==='reset'){
      await updateDoc(doc(db,'clients',clientCur.id),{visites:0,dates:[]});
      clientCur.visites=0; clientCur.dates=[];
      const ok=document.getElementById('msg-reset-ger');
      ok.style.display='block'; setTimeout(()=>ok.style.display='none',2000);
    }
    afficherGerant(clientCur);
  } catch(e){console.error(e);}
};

document.getElementById('pin-inp').addEventListener('keyup',e=>{if(e.key==='Enter')window.valPin();});
